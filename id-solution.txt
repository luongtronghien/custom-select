this._id = count++;

      initStructure(that);
      loadText(that);
      createDropdown(that);
      wrapSL = element.parent();
      ulDrop = that.ulDrop;

      changepositionDropdown();

      $window.on('click.' + pluginName, function (e, id) {
        if (that._id !== id) {
          that.hideDropDown();
        }
      });

      wrapSL.on('click.sel', function(e) {
        e.preventDefault();
        e.stopPropagation();


        $window.trigger('click.' + pluginName, that._id);
        changepositionDropdown();

        if (ulDrop.hasClass(hiddenClass)) {
          that.showDropDown();
        }
        else {
          console.log('hide');
          that.hideDropDown();
        }

      });
